ATOI на вход получает строку, с записанным в нее числовым значением. Возвращает переменную типа int.

int		ft_atoi(char *str)
{
	int		d;
	int		is_neg;

	is_neg = 1;
	d = 0;
	while (*str == ' ' || *str == '\n' || *str == '\t')
		str++;
	if (*str == '-')
	{
		is_neg = -1;
		str++;
	}
	while (*str >= '0' && *str <= '9')
	{
		d *= 10;
		d += (*str++) - '0';
		if (d == 2147483647)
			return (2147483647);
		if (d == 214748364 && *str == '8' && is_neg == -1)
			return (-2147483648);
	}
	return (is_neg * d);
}

int		ft_factorial(int nb)
{
	if (nb < 0 || nb == 12)
		return (0);
	if (nb == 0)
		return (1);
	return (nb * ft_factorial(nb - 1));
}

int ft_power(int nb, int power)
{
	if (power < 0)
		return (0);
	if (power == 0)
		return (1);
	return (nb * ft_power(nb, power - 1));
}

void	ft_putnbr(int nb)
{
	if (nb == 0)
	{
		ft_putchar('0');
		return ;
	}
	if (nb / 10 != 0)
		ft_putnbr(nb / 10);
	if (nb / 10 == 0 && nb < 0)
		ft_putchar('-');
	ft_putchar((nb > 0 ? nb % 10 : -(nb % 10)) + '0');
}

int		ft_sqrt(int nb)
{
	int i;

	i = 1;
	if (nb <= 0)
		return (0);
	while (i * i < nb)
		i++;
	if (i * i == nb)
		return (i);
	return (0);
}

int		ft_strcmp(char *s1, char *s2)
{
	while (*s1 == *s2 && *s1 != '\0' && *s2 != '\0')
	{
		s1++;
		s2++;
	}
	if (*s1 == '\0' && *s2 == '\0')
		return (0);
	else if (*s1 > *s2)
		return (1);
	else
		return (-1);
}

int		ft_strncmp(char *s1, char *s2, unsigned int n)
{
	while (*s1 == *s2 && *s1 != '\0' && *s2 != '\0' && --n > 0)
	{
		s1++;
		s2++;
	}
	if ((*s1 == '\0' && *s2 == '\0') || n == 0)
		return (0);
	else if (*s1 > *s2 || *s2 == '\0')
		return (1);
	else
		return (-1);
}

char	*ft_strstr(char *str, char *to_find)
{
	char *begin_f;
	char *pos;

	begin_f = to_find;
	while (*str != 0)
	{
		if (*str == *to_find)
		{
			pos = str;
			while (*pos++ == *to_find++ && *to_find != '\0')
				;
			if (*to_find == '\0')
				return (str);
			to_find = begin_f;
		}
		str++;
	}
	return (0);
}
